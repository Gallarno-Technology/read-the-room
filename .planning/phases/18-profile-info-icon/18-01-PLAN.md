```javascript
    function openInfo() {
      const info = PROFILE_INFO[activeProfile] || PROFILE_INFO['kids_present'];
      infoHeading.textContent = info.name;
      infoBody.textContent    = info.desc;
      infoBtn.setAttribute('aria-expanded', 'true');
      if (isMobile()) {
        // Mobile: class-based show (avoids hidden/transition conflict — Pitfall 1)
        infoPanel.removeAttribute('hidden');
        requestAnimationFrame(function() {
          infoPanel.classList.add('info-panel--open');
        });
        infoBackdrop.classList.add('active');
      } else {
        // Desktop: simple hidden toggle (no transition needed)
        infoPanel.removeAttribute('hidden');
        infoBackdrop.classList.remove('active');
      }
    }

    function closeInfo() {
      infoBtn.setAttribute('aria-expanded', 'false');
      if (isMobile()) {
        infoPanel.classList.remove('info-panel--open');
        infoBackdrop.classList.remove('active');
        // Restore hidden after slide-down transition completes (250ms)
        setTimeout(function() {
          if (!infoPanel.classList.contains('info-panel--open')) {
            infoPanel.setAttribute('hidden', '');
          }
        }, 260);
      } else {
        infoPanel.setAttribute('hidden', '');
        infoBackdrop.classList.remove('active');
      }
    }
```
