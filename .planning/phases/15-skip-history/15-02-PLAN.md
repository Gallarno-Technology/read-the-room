    **3. Add `hydrateFeed()` async function (per D-05, D-06, D-07).**

    Place after `prependWarningItem`:
    ```javascript
    async function hydrateFeed() {
      try {
        const resp = await fetch('/feed');
        const events = await resp.json();
        const existing = new Set();
        skipFeed.querySelectorAll('li[data-event-id]').forEach(function(li) {
          existing.add(Number(li.dataset.eventId));
        });
        var toRender = events.filter(function(e) { return !existing.has(e.id); }).reverse();
        for (var i = 0; i < toRender.length; i++) {
          var evt = toRender[i];
          if (evt.type === 'skip') {
            prependSkipItem(evt);
          } else if (evt.type === 'five_skip_warning') {
            prependWarningItem(evt);
          }
        }
      } catch (err) {
        // Silently fail -- SSE will pick up live events
      }
    }
    ```
