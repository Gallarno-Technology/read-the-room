```javascript
function setEvalBadge(evalState, severity) {
  nowPlayingBadge.className = 'badge ' + (EVAL_BADGE_CLASS[evalState] || 'badge--evaluating');
  nowPlayingBadge.textContent = EVAL_BADGE_LABEL[evalState] || '';
  // Remove any existing mild-language badge before potentially re-adding
  const existing = badgeGroup.querySelector('.badge--mild-language');
  if (existing) existing.remove();
  // Only render mild-language badge when track passed with severity >= 1 (D-07, D-09)
  if (evalState === 'passed' && severity >= 1) {
    const mildBadge = document.createElement('span');
    mildBadge.className = 'badge badge--mild-language';
    mildBadge.textContent = 'Mild language';
    badgeGroup.appendChild(mildBadge);
  }
}
```
