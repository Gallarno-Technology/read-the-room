    fsmMainZone.addEventListener('click', async function() {
      const newState = !fsmEnabled;
      setFsmUI(newState);
      fsmError.textContent = '';
      try {
        const resp = await fetch('/fsm', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ enabled: newState })
        });
        if (!resp.ok) {

    // ▾ dropdown trigger — right zone (D-05: stopPropagation prevents FSM toggle)
    profileTrigger.addEventListener('click', function(e) {

    // Profile option selection (D-09: saves to state.json, updates label, closes dropdown, no FSM change)
    profileDropdown.addEventListener('click', async function(e) {
      const opt = e.target.closest('.profile-option');
      if (!opt) return;
      const newProfile = opt.dataset.profile;
      if (!newProfile) return;

      const prevProfile = activeProfile;
      setFsmUI(fsmEnabled, newProfile);
      profileDropdown.hidden = true;
      profileTrigger.setAttribute('aria-expanded', 'false');
      fsmError.textContent = '';

      try {
        const resp = await fetch('/profile', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ profile: newProfile })
        });
        if (!resp.ok) {
          throw new Error('HTTP ' + resp.status);
        }
      } catch (err) {
        setFsmUI(fsmEnabled, prevProfile);
        fsmError.textContent = 'Could not save profile \u2014 try again.';
        setTimeout(function() { fsmError.textContent = ''; }, 3000);
      }
    });

    // Close dropdown on outside click (D-10)
    document.addEventListener('click', function(e) {
      if (!profileDropdown.hidden &&
          !profileDropdown.contains(e.target) &&
          e.target !== profileTrigger) {
        profileDropdown.hidden = true;
        profileTrigger.setAttribute('aria-expanded', 'false');
      }
    });

    // Close dropdown on Escape (D-10)
    document.addEventListener('keydown', function(e) {

        profileDropdown.hidden = true;
        profileTrigger.setAttribute('aria-expanded', 'false');
        profileTrigger.focus();
      }
    });
